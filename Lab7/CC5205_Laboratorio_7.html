# P1 - Proporcin de ceros y columnas que slo contienen ceros
import pandas as pd
df = pd.DataFrame(X)
# proporcin total de ceros
prop_zero = (df == 0).sum().sum() / df.size
# fraccin de ceros por columna
col_zero_frac = (df == 0).sum() / df.shape[0]
all_zero_cols = col_zero_frac[col_zero_frac == 1.0].index.tolist()
print(f'Proporcin de ceros en el dataset: {prop_zero:.4f}')
print(f'Nmero de columnas que son todo ceros: {len(all_zero_cols)}')
if len(all_zero_cols) > 0:
    print('Columnas (ndices) con slo ceros:', all_zero_cols)
# Guardar resultado en X2 (mantener X sin modificar). Eliminamos columnas todo-cero si existen
if len(all_zero_cols) > 0:
    X2 = df.drop(columns=all_zero_cols).values
else:
    X2 = X.copy()
# imprimir forma resultante
print('Shape original X:', X.shape)
print('Shape X2 despus de eliminar columnas todo-cero (si hubo):', X2.shape)

# P2 - Matriz de correlacin (muestra de subconjunto y reduccin por alta correlacin)
import numpy as np
import pandas as pd
df = pd.DataFrame(X2)
# matriz de correlacin (abs) - trabajamos con un subconjunto para visualizar
corr = df.corr().abs()
plt.figure(figsize=(10,8))
sns.heatmap(corr.iloc[:30, :30], cmap='vlag')
plt.title('Matriz de correlacin (subset primera 30 columnas)')
plt.show()
# detectar pares altamente correlacionados y eliminar una de cada par (umbral 0.95)
upper = corr.where(np.triu(np.ones(corr.shape), k=1).astype(bool))
to_drop = [column for column in upper.columns if any(upper[column] > 0.95)]
print('Nmero de columnas con correlacin alta (>0.95) respecto a otra:', len(to_drop))
if len(to_drop) > 0:
    print('Primeras 20 columnas a eliminar (ndices):', to_drop[:20])
# actualizar X2 eliminando columnas redundantes
if len(to_drop) > 0:
    X2 = df.drop(columns=to_drop).values
    print('Shape X2 luego de eliminar columnas altamente correlacionadas:', X2.shape)
else:
    print('No se eliminaron columnas por correlacin alta.')

# P6 - Entrenar y evaluar SVC para dataset original y reducciones
from sklearn.svm import SVC
from sklearn.metrics import confusion_matrix
import time
# Asegurar que las particiones estn definidas para cada conjunto:
# X_train/X_test ya vienen del split sobre X (sin reduccin)
# Recalcular divisin para X2 (versin final tras preprocesos)
X2_train, X2_test, y_train, y_test = train_test_split(X2, y, test_size=0.3, stratify=y, random_state=0)
# X3_train/X3_test y X4_train/X4_test ya se definieron en P4/P5
def run_svc(name, Xtr, Xte, ytr, yte):
    t0 = time.perf_counter()
    clf = SVC()
    clf.fit(Xtr, ytr)
    elapsed = time.perf_counter() - t0
    ypred = clf.predict(Xte)
    print(f'{name}  tiempo entrenamiento: {elapsed:.4f} s')
    plot_confusion_matrix(yte, ypred)
    return elapsed

# Ejecutar para cada conjunto:
